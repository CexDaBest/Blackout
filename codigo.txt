    // Lightbox para im√°genes (incluye: capturas, pasos y post-descarga)
    (function () {
      const lightbox = document.getElementById('lightbox');
      const lightboxImage = document.getElementById('lightboxImage');
      const lightboxCaption = document.getElementById('lightboxCaption');
      const closeBtn = document.getElementById('lightboxClose');

      const zoomable = document.querySelectorAll('.imagenes-abajo img, .steps .bubble img, #postDownloadGuide img');

      function openLightbox(src, alt, caption) {
        lightboxImage.src = src;
        lightboxImage.alt = alt || '';
        lightboxCaption.textContent = caption || alt || '';
        lightbox.setAttribute('aria-hidden', 'false');
        document.body.style.overflow = 'hidden';
        closeBtn.focus();
      }

      function closeLightbox() {
        lightbox.setAttribute('aria-hidden', 'true');
        lightboxImage.src = '';
        lightboxImage.alt = '';
        lightboxCaption.textContent = '';
        document.body.style.overflow = '';
      }

      zoomable.forEach(img => {
        img.addEventListener('click', () => {
          const src = img.src;
          const alt = img.alt || '';
          const caption = img.dataset.caption || alt || '';
          openLightbox(src, alt, caption);
        });
        img.setAttribute('tabindex', '0');
        img.style.cursor = 'zoom-in';
        img.addEventListener('keydown', (e) => {
          if (e.key === 'Enter' || e.key === ' ') {
            e.preventDefault();
            img.click();
          }
        });
      });

      lightbox.addEventListener('click', (e) => {
        if (e.target === lightbox || e.target === closeBtn) {
          closeLightbox();
        }
      });

      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && lightbox.getAttribute('aria-hidden') === 'false') {
          closeLightbox();
        }
      });

      const content = document.querySelector('.lightbox-content');
      content.addEventListener('click', (e) => { e.stopPropagation(); });

      let startY = null;
      lightbox.addEventListener('touchstart', (e) => {
        if (e.touches && e.touches.length === 1) startY = e.touches[0].clientY;
      });
      lightbox.addEventListener('touchend', (e) => {
        if (!startY) return;
        const endY = (e.changedTouches && e.changedTouches[0].clientY) || 0;
        if (endY - startY > 100) { closeLightbox(); }
        startY = null;
      });
    })();

        if (window.__resetBlackoutCounts) {

      }

        const input = prompt('Introduce PIN:');
        if (input === null) return; // cancelado

          triggerReset();

          showToast('PIN incorrecto');
        }
      });

            triggerReset();
          } else {
            showToast('PIN incorrecto');
          }
        }
      });
